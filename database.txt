create database job;
use job;

create table graduate_register(applicant_name varchar(40) primary key,
password varchar(40),
contact_no varchar(20),
email varchar(100), 
skills varchar(180),
resume varchar(45));

create table company_register(company_name varchar(50) primary key,
password varchar(40),
email varchar(50),
project_type varchar(20),
skills_required varchar(180));

create table postproject(project_id int primary key,
company_name varchar(50),
application_type varchar(30),
project_description varchar(250),
required_skills varchar(180),
posted_date varchar(30),
status varchar(40));
